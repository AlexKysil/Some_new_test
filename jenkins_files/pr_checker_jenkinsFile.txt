#!groovy
properties([disableConcurrentBuilds()])

pipeline {
    agent any
    options {
        timestamps()
    }
    stages {
        stage("Run static code analyzer") {
        steps {
            echo "<========== Running Pylint ==========>"
            sh 'find . -type f -name "*.py" | xargs pylint --reports=y text'
            }
        }
    }
}
